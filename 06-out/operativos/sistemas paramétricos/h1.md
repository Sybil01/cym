1. Descripción axiomática

El horno-órgano de barro se presenta como una estructura híbrida donde confluyen materia cerámica, energía térmica y dinámica acústica. Cada chimenea actúa como objeto resonador ($\obj_i$), excitado por el calor ($\mat_p$) y el flujo de aire que opera como agente causal ($\agn_c$). La instalación se define como interfaz performativa ($\itf$) entre fuego, humo y sonido, produciendo un campo donde los límites entre cocción, música y ritual se superponen. Se inscribe en la lógica de la distinción de Spencer-Brown y la diagramática de Peirce, donde un límite material se convierte en operador simbólico y sonoro ￼ ￼ ￼.

2. Fórmula de escritura icónica

$$
\newcommand{\mat}{\blacksquare}
\newcommand{\obj}{\blacklozenge}
\newcommand{\agn}{\bullet}
\newcommand{\itf}{\leftrightarrow}
\newcommand{\ent}[1]{\boxed{#1}}
$$

Fórmula simple:
$$
\mat_p \rightarrow \obj_i \itf \agn_c
$$

Fórmula compleja:
$$
(\mat_p \diamond^{-1} \mat_h) \rightarrow (\obj_i \cup \obj_s) \itf (\agn_h \oplus \agn_c)
$$

⸻

3. Desglose de términos
	•	$\mat_p$: material físico, el barro como soporte térmico y resonante.
	•	$\mat_h$: material híbrido, barro + aire caliente, mediación entre sólido y flujo.
	•	$\obj_i$: objeto instrumental, chimenea entonada como tubo de órgano.
	•	$\obj_s$: objeto sintético, configuración escultórica-acústica total.
	•	$\agn_c$: agente colectivo (fuego, viento, público), que activa el sistema.
	•	$\agn_h$: agente humano que enciende, sopla o modula.
	•	$\rightarrow$: causalidad, transformación del calor en presión sonora.
	•	$\itf$: interfaz, zona de contacto entre materia, aire y escucha.
	•	$\cup$: unión, coexistencia de varios tubos/chimeneas en un mismo sistema.
	•	$\oplus$: composición de agentes, interacción del humano y del fuego.
	•	$\diamond^{-1}$: modalización de paso, digitalización o virtualización de la materia.

4. Ejemplo en JavaScript (DataviewJS/Obsidian)

```dataviewjs
// Horno-órgano en 3D con nodos y sonido
(async () => {
  if (!window.THREE) {
    await new Promise(r => {
      const s = document.createElement("script");
      s.src = "https://cdn.jsdelivr.net/npm/three@0.150.0/build/three.min.js";
      s.onload = r;
      document.head.appendChild(s);
    });
  }

  const {Scene, PerspectiveCamera, WebGLRenderer, SphereGeometry, MeshBasicMaterial, Mesh, LineBasicMaterial, BufferGeometry, Line, Vector3} = THREE;

  const wrap = this.container;
  wrap.style.height = "400px";

  const scene = new Scene();
  const camera = new PerspectiveCamera(45, wrap.clientWidth / wrap.clientHeight, 0.1, 1000);
  camera.position.z = 10;

  const renderer = new WebGLRenderer();
  renderer.setSize(wrap.clientWidth, wrap.clientHeight);
  wrap.appendChild(renderer.domElement);

  const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  const gain = audioCtx.createGain();
  gain.gain.value = 0.2;
  gain.connect(audioCtx.destination);

  // Oscillator node (chimenea-agente)
  const osc = audioCtx.createOscillator();
  osc.type = "sine";
  osc.frequency.value = 220;
  osc.connect(gain);

  // Nodos esféricos (chimeneas y agentes)
  const nodes = [];
  const geom = new SphereGeometry(0.2, 16, 16);
  const matHuman = new MeshBasicMaterial({color: 0x3399ff});
  const matFire = new MeshBasicMaterial({color: 0xff3300});
  const matPipe = new MeshBasicMaterial({color: 0xaaaaaa});

  const human = new Mesh(geom, matHuman);
  human.position.set(-2, 0, 0);
  scene.add(human);
  nodes.push(human);

  const fire = new Mesh(geom, matFire);
  fire.position.set(0, 0, 0);
  scene.add(fire);
  nodes.push(fire);

  for (let i = 0; i < 3; i++) {
    const pipe = new Mesh(geom, matPipe);
    pipe.position.set(2, (i - 1) * 1.2, 0);
    scene.add(pipe);
    nodes.push(pipe);
  }

  // Líneas entre nodos
  function connect(a, b) {
    const material = new LineBasicMaterial({color: 0xffffff});
    const geometry = new BufferGeometry().setFromPoints([a.position, b.position]);
    const line = new Line(geometry, material);
    scene.add(line);
  }
  connect(human, fire);
  nodes.slice(2).forEach(p => connect(fire, p));

  // Interacción
  let running = false;
  const stateBtn = document.createElement("button");
  stateBtn.textContent = "Start";
  stateBtn.onclick = () => {
    if (!running) {osc.start(); running = true; stateBtn.textContent = "Stop";}
    else {osc.stop(); running = false; stateBtn.textContent = "Start";}
  };
  wrap.appendChild(stateBtn);

  // Hover cambia frecuencia
  renderer.domElement.addEventListener("mousemove", e => {
    const x = (e.offsetX / wrap.clientWidth - 0.5) * 2;
    osc.frequency.value = 220 + x * 220;
  });

  // Resize
  new ResizeObserver(() => {
    camera.aspect = wrap.clientWidth / wrap.clientHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(wrap.clientWidth, wrap.clientHeight);
  }).observe(wrap);

  function animate() {
    requestAnimationFrame(animate);
    renderer.render(scene, camera);
  }
  animate();
})();
```
